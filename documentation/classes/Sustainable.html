import {
  IsArray,
  IsBoolean,
  IsDate,
  IsDateString,
  IsDecimal,
  IsEnum,
  IsNotEmpty,
  IsNumber,
  IsNumberString,
  IsOptional,
  IsString,
  MaxLength,
} from 'class-validator';
import { Optional } from '@nestjs/common';

export class Sustainable {
  isActive: boolean;
  @MaxLength(300, {

  @IsArray()
  categories: Array<string>;

  @IsString()

  title: string;

  @IsString()

  })
  description: string;

  @IsString()

  @IsArray()

  @IsArray()

}
